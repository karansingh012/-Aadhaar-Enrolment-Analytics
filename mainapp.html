import pandas as pd
import re

def clean_state_name(x):
    if pd.isna(x):
        return x 
    x = str(x).lower()
    x = re.sub(r'[^a-z]', '', x) # remove symbols like &,
    x = re.sub(r'\s+',' ', x).strip() # remove extra spaces
    return x

import pandas as pd
import re

def clean_name(x):
    if pd.isna(x):
        return x 
    x = str(x).lower()
    x = re.sub(r'[^a-z]', '', x) 
    x = re.sub(r'\s+',' ', x).strip()
    return x
